int lexicographic_sort(const char* a, const char* b) {
    return strcmp(a,b);
}
void string_sort(char** arr,const int len,int (*cmp_func)(const char* a, const char* b)){
    int flag=0,top=len-1;
    while(!flag)
    {
        flag=1;
        for(int i=0;i<top;i++)
            if(cmp_func(arr[i],arr[i+1])>0)
            {
                char *t=arr[i];
                arr[i]=arr[i+1];
                arr[i+1]=t;
                flag=0;
            }
        top--;
    }
}

int no_of_ditinct_char_in_a_string(const char *a)
{	int n = 0;
    int count[128] = {0};
    if (NULL == a)
        return -1;
    while(*a != '\0')
    {  if (!count[*a])
	{count[*a]++;
         n++;
        }
        a++;
    }
    return n;
}
